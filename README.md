clone o projeto e baixe as dependencias. comandos:


git clone https://github.com/gavasarcenio-crypto/rickandmorty-wikiPam.git
cd .\rickandmorty-wikiPam\
npm install expo
npm install @react-navigation/native
npx expo install react-native-screens react-native-safe-area-context
npm install @react-navigation/stack
npm install axios
npx expo start --tunnel
